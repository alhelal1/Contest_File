int data[35]= {10000000,8, 1, 9, 8, 3, 4, 6, 1, 5, 2};
int I[30],L[35],print[35],n;

int LIS_n_logk()
{
    int length=0;
    int ans=0;
    for(int i=1; i<=n; i++)
    {
        int left=0,right=length,mid;
        while(left<=right)
        {
            mid=(left+right)/2;
            if(I[mid]<data[i])
                left=mid+1;
            else
                right=mid-1;
        }
        I[left]=data[i];
        L[i]=left;
        ans=max(ans,L[i]);
        if(left>length)
            length++;
    }
    return ans;
}

void print_LIS(int max_LIS)
{
    int nn=max_LIS;
    int top,i;
    for(i=n; i>0; i--)
        if(L[i]==nn)
        {
            top=data[i];break;
        }
    print[nn-1]=top;
    nn--;
    for(int j=i-1; j>0; j--)
    {
        if(L[j]==nn && data[j]<top)
        {
            print[--nn]=data[j];
            top=data[j];
        }
    }
    for(int i=0; i<max_LIS; i++)
        cout<<print[i]<<" ";
    cout<<endl;
}

long long mod; //     

class Matrix 
{
public:

    int row, col; // Matirx  row  column  
    long long valu[5][5]; // Matirx  


    void clear()  // Matirx    0   
    {
        memset(valu,0,sizeof(valu));
        row=0;
        col=0;
    }

 

    void identify(int x) // Identity Matrix  1 Matrix  
    {
        clear();
        row=x;
        col=x;

        for(int i=0; i<row; i++)
            valu[i][i]=1; //    1   
    }

 

    Matrix operator * (Matrix b) //   Matrix    Matrix   
    {
        int i,j,k;
        Matrix c;

        c.clear();
        c.row=row;
        c.col=b.col;
                               // Complexity d^3.           
                               // Matrix     x 

        for(i=0; i<row; i++)
            for(j=0; j<b.col; j++)
                for(k=0; k<col; k++)
                {
                    c.valu[i][j] = c.valu[i][j] + (valu[i][k])*(b.valu[k][j]);
                    c.valu[i][j] %= mod;
                }
        return c;
    }

 

    Matrix operator ^ (long long p)   // Fast Exponential for Matrix, Matrix^p  
    {
        Matrix res, x;
        res.identify(row);

        x=*(this);

        while(p)
        {
            if(p & 1) res = res * x;
            x=x*x;
            p>>=1;
        }
        return res;

    }

};

int mod=1000;
int row;
 
struct Matrix
{
    ll mat[10][10];
};
 
Matrix Mul(Matrix aa, Matrix bb)
{
    Matrix ret;
    cover(ret.mat,0);
    for(int i = 0; i < row; i ++)
    {
        for(int j = 0; j < row; j ++)
        {
            for(int k = 0; k < row; k ++)
            {
                ret.mat[i][j] += (aa.mat[i][k]*bb.mat[k][j]);
                ret.mat[i][j] %= mod;
            }
        }
    }
    return ret;
}
 
Matrix exp(Matrix a, ll b)
{
    if(b == 1) return a;
    if(b&1)
    {
        return Mul(a, exp(a,b-1));
    }
    else
    {
        Matrix tmp = exp(a,b/2);
        return Mul(tmp,tmp);
    }
}

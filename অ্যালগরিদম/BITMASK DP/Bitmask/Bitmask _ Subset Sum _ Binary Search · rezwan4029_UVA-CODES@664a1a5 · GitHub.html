+vl PowerSet(vl a){
+ int len = a.size();
+ vl S;
+ rep(mask,(1<<len)){
+ ll sum = 0 ;
+ rep(i,len){
+ if(mask & (1<<i)) sum += a[i];
+ }
+ S.pb(sum);
+ }
+ return S;
+}

#include<bits/stdc++.h>
using namespace std;
 
int a[1000099];
void sieve_for_sum_of_distinct_prime_factor()
{
    long long i,j;
    for(i=2;i<=1000000;i++)
    {
        if(a[i]==0)
        {
            for(j=i;j<=1000000;j=j+i)
                a[j]=a[j]+i;
        }
    }
}
 
int main()
{
    int i,j,k,n,m,d;
    sieve_for_sum_of_distinct_prime_factor();
    for(i=1;i<=20;i++)
        printf("Sum of distinct prime factor of %d is %d\n",i,a[i]);
}
